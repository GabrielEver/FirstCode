\section{Primeira Questão}
\begin{definition}


    Aqui temos um Sistema Linear, e colocaremos o Sistema em formar Matricial, ou seja, A\cdot x = b.
    
\newline
\begin{center}
    A = $\left\{\begin{array}{lllll}
    +x\ \ +y\ \ +2z\ \ +w\ \ = \ \ 22\\\\ +x\ \ -y\ \ -z\ \ +2w\ \ = \ \ 20\\\\ -x\ \ -2y \ \ +z \ \ +2w\ =\ \ 11 \\\\ +x\ \ +y \ \ +z\ \ +w \ \ = \ \ 20
    \end{array}\right\\$
\newline
\end{center}
\end{definition}

%Resultado
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
